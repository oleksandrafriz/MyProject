import React from "react";
import app from "../firebase";
import { getDatabase, ref, get } from "firebase/database";
import { useNavigate } from "react-router-dom";
import { useState, useEffect } from "react";
import styles from "./Card.module.css";
import SearchBar from "./SearchBar";

/**
 * Компонент для відображення списку мультфільмів із бази даних Firebase.
 * @returns {JSX.Element} Компонент списку мультфільмів.
 */
function Card() {
  /**
   * Стан, що зберігає відфільтрований список мультфільмів.
   * @type {Array<Object>}
   */
  const [movieArray, setMovieArray] = useState([]);

  /**
   * Стан для збереження пошукового запиту.
   * @type {string}
   */
  let [searchedCartoon, setSearchedCartoon] = useState("");

  /**
   * Стан для збереження оригінального списку мультфільмів (без фільтрації).
   * @type {Array<Object>}
   */
  const [originalMovieArray, setOriginalMovieArray] = useState([]);
  const navigate = useNavigate();

  /**
   * Використання useEffect для отримання даних з Firebase при завантаженні компоненту.
   * @returns {void}
   */
  useEffect(() => {
    /**
     * Асинхронна функція для отримання даних з бази даних Firebase.
     * @async
     * @function fetchData
     * @returns {Promise<void>}
     */
    const fetchData = async () => {
      try {
        const db = getDatabase(app);
        const dbRef = ref(db, "cartoons");
        const snapshot = await get(dbRef);

        if (snapshot.exists()) {
          const data = snapshot.val();
          const dataArray = Object.entries(data).map(([id, cartoon]) => ({
            id,
            ...cartoon,
          }));
          setMovieArray(dataArray);
          setOriginalMovieArray(dataArray);
        } else {
          alert("error");
        }
      } catch (error) {
        console.error("Помилка отримання даних:", error);
      }
    };

    fetchData();
  }, []);

  /**
   * Функція для фільтрації мультфільмів за пошуковим запитом.
   *
   * @param {React.ChangeEvent<HTMLInputElement>} e - Подія введення тексту.
   * @returns {void}
   */
  const handleSearch = (e) => {
    const searchCartoon = e.target.value;
    setSearchedCartoon(searchCartoon);

    if (searchCartoon === "") {
      setMovieArray(originalMovieArray);
      return;
    }
    const filterBySearch = originalMovieArray.filter((item) => {
      return item.title.toLowerCase().includes(searchCartoon.toLowerCase());
    });
    setMovieArray(filterBySearch);
  };

  /**
   * Функція для обробки кліку по мультфільму та переходу на сторінку деталей.
   *
   * @param {string} id - Ідентифікатор мультфільму.
   * @returns {function(): void} Функція, що виконує навігацію до сторінки деталей.
   */
  const handleClick = (id) => {
    return () => navigate(`/cartoon/${id}`);
  };

  return (
    <>
      <SearchBar
        searchedCartoon={searchedCartoon}
        searchChange={handleSearch}
      />

      <div className={styles.cardContainer}>
        {movieArray.map((cartoon) => (
          <CardItem
            key={cartoon.id}
            id={cartoon.id}
            title={cartoon.title}
            year={cartoon.year}
            image={cartoon.image}
            onClick={handleClick(cartoon.id)}
          />
        ))}
      </div>
    </>
  );
}

export default Card;
